```yaml
- name: 戬心Press # 昵称
  desc: 致敬每个爱他们的你。 # 介绍
  avatar: https://jx.alloy.ink/img/avater.jpg # 头像
  link: /categories/?category=戬心Press # 可选，默认为指向昵称分类
  bgColor: '#CBEAFA' # 可选，默认var(--bodyBg)随主题改变。颜色值有#号时请添加单引号
```
